# Página no encontrada

La página que buscas no existe o ha sido movida.

Te invito a leer alguno de los siguientes artículos:

```{postlist} 5
:date: %Y-%m-%d
:format: {date} - {title}
:excerpts:
```

<script>
(function () {
  "use strict";
  var path = window.location.pathname.replace(/\/+$/, "").toLowerCase();
  var lookup = path + "/";

  fetch("/redirects.json")
    .then(function (r) { return r.ok ? r.json() : {}; })
    .then(function (map) {
      /* 1. Try redirect map (legacy blog post paths) */
      if (map[lookup]) {
        window.location.replace(map[lookup]);
        return;
      }

      /* 2. Detect tag/category pattern → search */
      var match = path.match(/\/(?:category|tag)\/([^/]+)/);
      if (match) {
        var query = decodeURIComponent(match[1]).replace(/-/g, " ");
        window.location.replace("/search/?q=" + encodeURIComponent(query));
        return;
      }

      /* 3. Default: recent posts already visible */
    })
    .catch(function () {
      /* Fetch failed: recent posts stay visible as fallback */
    });
})();
</script>
